// Funções de cache
function cacheData(key, data, ttl = 1800000) {
  const now = new Date().getTime();
  localStorage.setItem(key, JSON.stringify({ data, expiry: now + ttl }));
}
function getCachedData(key) {
  const cached = localStorage.getItem(key);
  if (cached) {
    try {
      const { data, expiry } = JSON.parse(cached);
      if (new Date().getTime() < expiry) return data;
    } catch (e) {
      console.error(`Erro ao parsear cache ${key}:`, e);
      localStorage.removeItem(key); // Limpa cache corrompido
    }
  }
  return null;
}

// Função para mostrar o conteúdo do banner

  document.getElementById('banner-content').style.display = 'grid';
}

// Função para atualizar o status de notícias
function updateNewsStatus(status, isError = false) {
  document.getElementById('news-status').innerHTML = `<i class="fas ${isError ? 'fa-exclamation-circle text-red-500' : 'fa-check-circle text-green-500'} mr-1"></i>${status}`;
}

// Função para buscar dados do clima (OpenWeatherMap API)
async function fetchWeather() {
  const cached = getCachedData('weather');
  if (cached) {
    document.getElementById('weather').innerHTML = cached;
    return;
  }
  const city = 'João Pessoa,BR';
  const url = `https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${API_KEYS.openWeatherMap}&units=metric&lang=pt_br`;
  try {
    const response = await fetch(url);
    const data = await response.json();
    if (data.cod === 200) {
      const html = `
        <p>${data.weather[0].description.charAt(0).toUpperCase() + data.weather[0].description.slice(1)}</p>
        <p>Temperatura: ${data.main.temp}°C</p>
        <p>Umidade: ${data.main.humidity}%</p>
        <p class="text-sm text-gray-500 mt-2">Atualizado em: ${new Date().toLocaleString('pt-BR')}</p>
      `;
      document.getElementById('weather').innerHTML = html;
      cacheData('weather', html);
    } else {
      console.error('Erro na API OpenWeatherMap:', data.message);
      document.getElementById('weather').innerHTML = `Erro: ${data.message || 'Não foi possível carregar o clima.'} Tente novamente mais tarde.`;
    }
  } catch (error) {
    console.error('Erro ao chamar OpenWeatherMap:', error);
    document.getElementById('weather').innerHTML = 'Não foi possível carregar o clima. Verifique sua conexão ou a chave da API.';
  }
}

// Função para buscar cotações de moedas (ExchangeRate-API)
async function fetchCurrency() {
  const cached = getCachedData('currency');
  if (cached) {
    document.getElementById('currency').innerHTML = cached;
    return;
  }
  const url = `https://v6.exchangerate-api.com/v6/${API_KEYS.exchangeRate}/latest/BRL`;
  try {
    const response = await fetch(url);
    const data = await response.json();
    if (data.result === 'success') {
      const html = `
        <p>USD/BRL: ${data.conversion_rates.USD.toFixed(2)}</p>
        <p>EUR/BRL: ${data.conversion_rates.EUR.toFixed(2)}</p>
        <p class="text-sm text-gray-500 mt-2">Atualizado em: ${new Date().toLocaleString('pt-BR')}</p>
      `;
      document.getElementById('currency').innerHTML = html;
      cacheData('currency', html);
    } else {
      console.error('Erro na API ExchangeRate:', data.error_type);
      document.getElementById('currency').innerHTML = 'Não foi possível carregar cotações. Tente novamente mais tarde.';
    }
  } catch (error) {
    console.error('Erro ao chamar ExchangeRate-API:', error);
    document.getElementById('currency').innerHTML = 'Não foi possível carregar cotações. Verifique sua conexão.';
  }
}

// Função para buscar notícias via RSS
async function fetchNews() {
  const cached = getCachedData('news');
  if (cached && Array.isArray(cached)) {
    updateNewsCarousel(cached);
    updateNewsStatus('Notícias carregadas do cache.');
    return;
  }
  const rssUrls = [
    'https://api.rss2json.com/v1/api.json?rss_url=https://g1.globo.com/pb/paraiba/rss',
    'https://api.rss2json.com/v1/api.json?rss_url=https://jornaldaparaiba.com.br/rss',
    'https://api.rss2json.com/v1/api.json?rss_url=https://www.vaticannews.va/pt.rss.xml',
    'https://api.rss2json.com/v1/api.json?rss_url=http://www.angop.ao/angola/en_us/noticias/feed.xml',
    'https://api.rss2json.com/v1/api.json?rss_url=https://www.mnnonline.org/feed/',
    'https://api.rss2json.com/v1/api.json?rss_url=https://www.christianaid.org.uk/rss'
  ];
  try {
    let allNews = [];
    for (const url of rssUrls) {
      console.log('Buscando RSS:', url);
      const response = await fetch(url);
      const data = await response.json();
      if (data.status === 'ok' && Array.isArray(data.items)) {
        const filteredNews = data.items
          .filter(item => {
            const text = (item.title + ' ' + (item.description || '') + ' ' + (item.content || '')).toLowerCase();
            const relevant = (text.includes('sertão') ||
                             text.includes('crianças') ||
                             text.includes('fome') ||
                             text.includes('desnutrição') ||
                             text.includes('morte') ||
                             text.includes('áfrica') ||
                             text.includes('angola') ||
                             text.includes('revolta') ||
                             text.includes('guerra') ||
                             text.includes('pobreza') ||
                             text.includes('desigualdade') ||
                             text.includes('carência') ||
                             text.includes('seca') ||
                             text.includes('violência') ||
                             text.includes('conflito') ||
                             text.includes('missão') ||
                             text.includes('caridade') ||
                             text.includes('humanitária') ||
                             text.includes('igreja') ||
                             text.includes('cristã') ||
                             text.includes('solidariedade')) &&
                            !text.includes('show') &&
                            !text.includes('festival') &&
                            !text.includes('cantor') &&
                            !text.includes('música') &&
                            !text.includes('concerto') &&
                            !text.includes('evento') &&
                            !text.includes('esporte') &&
                            !text.includes('artista') &&
                            !text.includes('show cultural') &&
                            !text.includes('apresentação') &&
                            !text.includes('teatro') &&
                            !text.includes('dança');
            return relevant;
          })
          .map(item => ({
            title: item.title,
            link: item.link,
            description: item.description || '',
            pubDate: new Date(item.pubDate || new Date())
          }));
        console.log(`Notícias filtradas de ${url}:`, filteredNews.length);
        allNews = allNews.concat(filteredNews);
      } else {
        console.error(`Erro no RSS ${url}:`, data.message || 'Falha ao carregar feed');
      }
    }
    allNews.sort((a, b) => b.pubDate - a.pubDate);
    console.log('Total de notícias encontradas:', allNews.length);
    if (allNews.length === 0) {
      updateNewsStatus('Nenhuma notícia encontrada nos feeds RSS. Tentando NewsAPI...', true);
      await fetchNewsFallback();
      return;
    }
    const newsData = allNews.slice(0, 5);
    updateNewsCarousel(newsData);
    cacheData('news', JSON.stringify(newsData));
    updateNewsStatus('Notícias carregadas com sucesso.');
  } catch (error) {
    console.error('Erro ao carregar notícias RSS:', error);
    updateNewsStatus('Erro ao carregar notícias RSS. Tentando NewsAPI...', true);
    await fetchNewsFallback();
  }
}

// Função para buscar notícias via NewsAPI (fallback)
async function fetchNewsFallback() {
  const query = '(pobreza OR crianças OR fome OR desnutrição OR morte OR revolta OR guerra OR seca OR violência OR conflito OR missão OR caridade OR humanitária OR igreja OR cristã OR solidariedade) AND (Sertão OR África OR Angola) -show -festival -cantor -música -concerto -evento -esporte -artista -show cultural -apresentação -teatro -dança';
  const url = `https://newsapi.org/v2/everything?q=${encodeURIComponent(query)}&sortBy=publishedAt&apiKey=${API_KEYS.newsApi}`;
  try {
    console.log('Buscando NewsAPI:', url);
    const response = await fetch(url);
    const data = await response.json();
    if (data.status === 'ok' && Array.isArray(data.articles)) {
      const newsData = data.articles
        .filter(article => {
          const text = (article.title + ' ' + (article.description || '')).toLowerCase();
          return !text.includes('show') && 
                 !text.includes('festival') && 
                 !text.includes('cantor') && 
                 !text.includes('música') && 
                 !text.includes('concerto') && 
                 !text.includes('evento') && 
                 !text.includes('esporte') && 
                 !text.includes('artista') &&
                 !text.includes('show cultural') &&
                 !text.includes('apresentação') &&
                 !text.includes('teatro') &&
                 !text.includes('dança');
        })
        .slice(0, 5)
        .map(article => ({
          title: article.title,
          link: article.url,
          description: article.description || '',
          pubDate: new Date(article.publishedAt)
        }));
      updateNewsCarousel(newsData);
      cacheData('news', JSON.stringify(newsData));
      updateNewsStatus('Notícias carregadas via NewsAPI.');
    } else {
      console.error('Nenhuma notícia encontrada na NewsAPI:', data.message || 'Sem resultados');
      document.getElementById('news').innerHTML = `<p>Nenhuma notícia encontrada no momento. Novas notícias serão exibidas em breve.</p>`;
      updateNewsStatus('Nenhuma notícia encontrada na NewsAPI.', true);
    }
  } catch (error) {
    console.error('Erro ao carregar notícias NewsAPI:', error);
    document.getElementById('news').innerHTML = `<p>Não foi possível carregar notícias. Verifique sua conexão.</p>`;
    updateNewsStatus('Erro ao carregar notícias NewsAPI.', true);
  }
}

// Função para atualizar o carrossel de notícias
function updateNewsCarousel(news) {
  const wrapper = document.querySelector('#news .swiper-wrapper');
  if (!Array.isArray(news) || news.length === 0) {
    wrapper.innerHTML = `<div class="swiper-slide"><p>Nenhuma notícia disponível no momento.</p></div>`;
    updateNewsStatus('Nenhuma notícia disponível.', true);
    new Swiper('#news', { pagination: { el: '.swiper-pagination' } });
    return;
  }
  wrapper.innerHTML = news.map(item => `
    <div class="swiper-slide">
      <p class="mb-2"><a href="${item.link}" target="_blank" class="text-blue-600 hover:underline">${item.title}</a></p>
      <p class="text-sm text-gray-600">${item.description ? item.description.substring(0, 100) + '...' : ''}</p>
    </div>
  `).join('');
  wrapper.insertAdjacentHTML('afterend', `<p class="text-sm text-gray-500 mt-2">Atualizado em: ${new Date().toLocaleString('pt-BR')}</p>`);
  new Swiper('#news', {
    pagination: { el: '.swiper-pagination' },
    navigation: {
      nextEl: '.swiper-button-next',
      prevEl: '.swiper-button-prev'
    },
    autoplay: { delay: 5000 },
    loop: news.length > 1
  });
}
